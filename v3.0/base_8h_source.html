530 LIGHTMODBUS_WARN_UNUSED LIGHTMODBUS_ALWAYS_INLINE static inline ModbusError modbusUnpackTCP(
531  const uint8_t *frame,
532  uint16_t length,
533  const uint8_t **pdu,
534  uint16_t *pduLength,
535  uint16_t *transactionID,
536  uint8_t *unitID)
537 {
538  // Check length
539  if (length < MODBUS_TCP_ADU_MIN || length > MODBUS_TCP_ADU_MAX)
540  return MODBUS_ERROR_LENGTH;
541 
542  // Discard non-Modbus messages
543  uint16_t protocolID = modbusRBE(&frame[2]);
544  if (protocolID != 0)
545  return MODBUS_ERROR_BAD_PROTOCOL;
546 
547  // Length mismatch
548  uint16_t messageLength = modbusRBE(&frame[4]);
549  if (messageLength != length - 6)
550  return MODBUS_ERROR_LENGTH;
551 
552  *transactionID = modbusRBE(&frame[0]);
553  *unitID = frame[6];
554  *pdu = frame + MODBUS_TCP_PDU_OFFSET;
555  *pduLength = length - MODBUS_TCP_ADU_PADDING;
556 
557  return MODBUS_OK;
