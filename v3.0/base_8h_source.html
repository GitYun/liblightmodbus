491 LIGHTMODBUS_WARN_UNUSED LIGHTMODBUS_ALWAYS_INLINE static inline ModbusError modbusUnpackTCP(
492  const uint8_t *frame,
493  uint16_t length,
494  const uint8_t **pdu,
495  uint16_t *pduLength,
496  uint16_t *transactionID,
497  uint8_t *unitID)
498 {
499  // Check length
500  if (length < MODBUS_TCP_ADU_MIN || length > MODBUS_TCP_ADU_MAX)
501  return MODBUS_ERROR_LENGTH;
502 
503  // Discard non-Modbus messages
504  uint16_t protocolID = modbusRBE(&frame[2]);
505  if (protocolID != 0)
506  return MODBUS_ERROR_BAD_PROTOCOL;
507 
508  // Length mismatch
509  uint16_t messageLength = modbusRBE(&frame[4]);
510  if (messageLength != length - 6)
511  return MODBUS_ERROR_LENGTH;
512 
513  *transactionID = modbusRBE(&frame[0]);
514  *unitID = frame[6];
515  *pdu = frame + MODBUS_TCP_PDU_OFFSET;
516  *pduLength = length - MODBUS_TCP_ADU_PADDING;
517 
518  return MODBUS_OK;
519 }
