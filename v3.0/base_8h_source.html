535 LIGHTMODBUS_WARN_UNUSED LIGHTMODBUS_ALWAYS_INLINE static inline ModbusError modbusUnpackTCP(
536  const uint8_t *frame,
537  uint16_t length,
538  const uint8_t **pdu,
539  uint16_t *pduLength,
540  uint16_t *transactionID,
541  uint8_t *unitID)
542 {
543  // Check length
544  if (length < MODBUS_TCP_ADU_MIN || length > MODBUS_TCP_ADU_MAX)
545  return MODBUS_ERROR_LENGTH;
546 
547  // Discard non-Modbus messages
548  uint16_t protocolID = modbusRBE(&frame[2]);
549  if (protocolID != 0)
550  return MODBUS_ERROR_BAD_PROTOCOL;
551 
552  // Length mismatch
553  uint16_t messageLength = modbusRBE(&frame[4]);
554  if (messageLength != length - 6)
555  return MODBUS_ERROR_LENGTH;
556 
557  *transactionID = modbusRBE(&frame[0]);
558  *unitID = frame[6];
559  *pdu = frame + MODBUS_TCP_PDU_OFFSET;
560  *pduLength = length - MODBUS_TCP_ADU_PADDING;
561 
562  return MODBUS_OK;
563 }
