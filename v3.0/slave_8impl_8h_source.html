296 LIGHTMODBUS_RET_ERROR modbusParseRequestRTU(ModbusSlave *status, uint8_t slaveAddress, const uint8_t *request, uint16_t requestLength)
297 {
298  // Unpack the request
299  const uint8_t *pdu;
300  uint16_t pduLength;
301  uint8_t requestAddress;
302  ModbusError err = modbusUnpackRTU(
303  request,
304  requestLength,
305  1,
306  &pdu,
307  &pduLength,
308  &requestAddress
309  );
310 
311  if (err != MODBUS_OK)
312  return MODBUS_MAKE_ERROR(MODBUS_ERROR_SOURCE_REQUEST, err);
313 
314  // Verify if the frame is meant for us
315  if (requestAddress != 0 && requestAddress != slaveAddress)
316  return MODBUS_REQUEST_ERROR(ADDRESS);
317 
318  // Parse the request
319  ModbusErrorInfo errinfo;
320  modbusBufferModeRTU(&status->response);
321  if (!modbusIsOk(errinfo = modbusParseRequest(status, pdu, pduLength)))
322  return errinfo;
323 
324  if (status->response.length)
325  {
326  // Discard any response frames if the request
327  // was broadcast
328  if (requestAddress == 0)
329  {
330  modbusSlaveFreeResponse(status);
331  return MODBUS_NO_ERROR();
332  }
333 
334  // Pack the response frame
335  err = modbusPackRTU(
336  &status->response.data[0],
337  status->response.length,
338  slaveAddress);
339 
340  if (err != MODBUS_OK)
341  return MODBUS_MAKE_ERROR(MODBUS_ERROR_SOURCE_GENERAL, err);
342  }
343 
344  return MODBUS_NO_ERROR();
