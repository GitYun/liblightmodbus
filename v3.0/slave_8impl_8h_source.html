245 LIGHTMODBUS_RET_ERROR modbusParseRequestRTU(ModbusSlave *status, const uint8_t *data, uint16_t length)
246 {
247  // Check length
248  if (length < 4 || length > 256)
249  return MODBUS_REQUEST_ERROR(LENGTH);
250 
251  // Check if the message is meant for us
252  uint8_t address = data[0];
253  if (address != status->address || address == 0)
254  return MODBUS_NO_ERROR();
255 
257 
258  // Check CRC
259  if (modbusCRC(data, length - 2) != modbusRLE(data + length - 2))
260  return MODBUS_REQUEST_ERROR(CRC);
261 
262  // Parse the request
263  ModbusErrorInfo err;
264  status->response.pduOffset = 1;
265  status->response.padding = 3;
266  if (!modbusIsOk(err = modbusParseRequest(status, address, data + 1, length - 3)))
267  return err;

272  status->response.data[0] = address;
273  modbusWLE(

290 LIGHTMODBUS_RET_ERROR modbusParseRequestTCP(ModbusSlave *status, const uint8_t *data, uint16_t length)
291 {
292  // Check length
293  if (length < 8 || length > 260)
294  return MODBUS_REQUEST_ERROR(LENGTH);
295 
296  // Read MBAP header
297  uint16_t transactionID = modbusRBE(&data[0]);
298  uint16_t protocolID = modbusRBE(&data[2]);
299  uint16_t messageLength = modbusRBE(&data[4]);
300  uint8_t address = status->address;
301 
302  // Discard non-Modbus messages
303  if (protocolID != 0)
304  return MODBUS_NO_ERROR();
305 
306  // Length mismatch
307  if (messageLength != length - 6)
308  return MODBUS_REQUEST_ERROR(LENGTH);
309 
310  // Parse the request
311  ModbusErrorInfo err;
312  status->response.pduOffset = 7;
313  status->response.padding = 7;
314  if (!modbusIsOk(err = modbusParseRequest(status, address, data + 7, messageLength - 1)))
315  return err;
316 
317  // Write MBAP header
318  if (status->response.length)
319  {
320  modbusWBE(&status->response.data[0], transactionID);
321  modbusWBE(&status->response.data[2], protocolID);
322  modbusWBE(&status->response.data[4], status->response.length - 6);
323  status->response.data[6] = address;
324  }
325 
326  return MODBUS_NO_ERROR();
