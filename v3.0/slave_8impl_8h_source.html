283 LIGHTMODBUS_RET_ERROR modbusParseRequestRTU(ModbusSlave *status, const uint8_t *request, uint16_t requestLength)
284 {
285  // Check length
286  if (requestLength < 4 || requestLength > 256)
287  return MODBUS_REQUEST_ERROR(LENGTH);
288 
289  // Check if the message is meant for us
290  uint8_t address = request[0];
291  if (address && address != status->address)
292  {
293  if (modbusSlaveAllocateResponse(status, 0))
294  return MODBUS_GENERAL_ERROR(ALLOC);
295  return MODBUS_NO_ERROR();
296  }
297 
298  // Check CRC
299  if (modbusCRC(request, requestLength - 2) != modbusRLE(request + requestLength - 2))
300  return MODBUS_REQUEST_ERROR(CRC);
301 
302  // Parse the request
303  ModbusErrorInfo err;
304  status->response.pduOffset = 1;
305  status->response.padding = 3;
306  if (!modbusIsOk(err = modbusParseRequest(status, address, request + 1, requestLength - 3)))
307  return err;
308 
309  // Write address and CRC to the reponse
310  if (status->response.length)
311  {
312  status->response.data[0] = address;
313  modbusWLE(
314  &status->response.data[status->response.length - 2],
315  modbusCRC(status->response.data, status->response.length - 2)
316  );
317  }
318 
319  return MODBUS_NO_ERROR();
320 }

333 LIGHTMODBUS_RET_ERROR modbusParseRequestTCP(ModbusSlave *status, const uint8_t *request, uint16_t requestLength)
334 {
335  // Check length
336  if (requestLength < 8 || requestLength > 260)
337  return MODBUS_REQUEST_ERROR(LENGTH);
338 
339  // Read MBAP header
340  uint16_t transactionID = modbusRBE(&request[0]);
341  uint16_t protocolID = modbusRBE(&request[2]);
342  uint16_t messageLength = modbusRBE(&request[4]);
343 
344  // Discard non-Modbus messages
345  if (protocolID != 0)
346  {
347  if (modbusSlaveAllocateResponse(status, 0))
348  return MODBUS_GENERAL_ERROR(ALLOC);
349  return MODBUS_NO_ERROR();
350  }
351 
352  // Length mismatch
353  if (messageLength != requestLength - 6)
354  return MODBUS_REQUEST_ERROR(LENGTH);
355 
356  // Parse the request
357  ModbusErrorInfo err;
358  status->response.pduOffset = 7;
359  status->response.padding = 7;
360  if (!modbusIsOk(err = modbusParseRequest(status, status->address, request + 7, messageLength - 1)))
361  return err;
362 
363  // Write MBAP header
364  if (status->response.length)
365  {
366  modbusWBE(&status->response.data[0], transactionID);
367  modbusWBE(&status->response.data[2], protocolID);
368  modbusWBE(&status->response.data[4], status->response.length - 6);
369  status->response.data[6] = request[6]; // Copy Unit ID
370  }
371 
372  return MODBUS_NO_ERROR();
373 }
