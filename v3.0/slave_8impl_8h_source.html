359 LIGHTMODBUS_RET_ERROR modbusParseRequestRTU(ModbusSlave *status, uint8_t slaveAddress, const uint8_t *request, uint16_t requestLength)
360 {
361  // Unpack the request
362  const uint8_t *pdu;
363  uint16_t pduLength;
364  uint8_t requestAddress;
365  ModbusError err = modbusUnpackRTU(
366  request,
367  requestLength,
368  1,
369  &pdu,
370  &pduLength,
371  &requestAddress
372  );
373 
374  if (err != MODBUS_OK)
375  return MODBUS_MAKE_ERROR(MODBUS_ERROR_SOURCE_REQUEST, err);
376 
377  // Verify if the frame is meant for us
378  if (requestAddress != 0 && requestAddress != slaveAddress)
379  return MODBUS_REQUEST_ERROR(ADDRESS);
380 
381  // Parse the request
382  ModbusErrorInfo errinfo;
383  status->response.pduOffset = MODBUS_RTU_PDU_OFFSET;
384  status->response.padding = MODBUS_RTU_ADU_PADDING;
385  if (!modbusIsOk(errinfo = modbusParseRequest(status, pdu, pduLength)))
386  return errinfo;
387 
388  if (status->response.length)
389  {
390  // Discard any response frames if the request
391  // was broadcast
392  if (requestAddress == 0)
393  {
394  modbusSlaveFreeResponse(status);
395  return MODBUS_NO_ERROR();
396  }
397 
398  // Pack the response frame
399  err = modbusPackRTU(
400  &status->response.data[0],
401  status->response.length,
402  slaveAddress);
403 
404  if (err != MODBUS_OK)
405  return MODBUS_MAKE_ERROR(MODBUS_ERROR_SOURCE_GENERAL, err);
406  }
407 
408  return MODBUS_NO_ERROR();
409 }
