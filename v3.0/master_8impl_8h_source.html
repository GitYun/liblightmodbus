195 LIGHTMODBUS_RET_ERROR modbusParseResponsePDU(
196  ModbusMaster *status,
197  uint8_t address,
198  const uint8_t *request,
199  uint8_t requestLength,
200  const uint8_t *response,
201  uint8_t responseLength)
202 {
203  // Check if lengths are ok
204  if (!requestLength || requestLength > MODBUS_PDU_MAX)
205  return MODBUS_REQUEST_ERROR(LENGTH);
206  if (!responseLength || responseLength > MODBUS_PDU_MAX)
207  return MODBUS_RESPONSE_ERROR(LENGTH);
208 
209  uint8_t function = response[0];
210 
211  // Handle exception frames
212  if (function & 0x80 && responseLength == 2)
213  {
214  if (status->exceptionCallback)
215  status->exceptionCallback(
216  status,
217  address,
218  function & 0x7f,
219  (ModbusExceptionCode) response[1]);
220 
221  return MODBUS_NO_ERROR();
222  }
223 
224  // Check if function code matches the one in request frame
225  if (function != request[0])
226  return MODBUS_RESPONSE_ERROR(FUNCTION);
227 
228  // Find a parsing function
229  for (uint16_t i = 0; i < status->functionCount; i++)
230  if (function == status->functions[i].id)
231  return status->functions[i].ptr(
232  status,
233  address,
234  function,
235  request,
236  requestLength,
237  response,
238  responseLength);
239 
240  // No matching function handler
241  return MODBUS_GENERAL_ERROR(FUNCTION);
242 }
