280 LIGHTMODBUS_RET_ERROR modbusParseResponsePDU(
281  ModbusMaster *status,
282  uint8_t address,
283  const uint8_t *request,
284  uint8_t requestLength,
285  const uint8_t *response,
286  uint8_t responseLength)
287 {
288  // Check if lengths are ok
289  if (!requestLength) return MODBUS_REQUEST_ERROR(LENGTH);
290  if (!responseLength) return MODBUS_RESPONSE_ERROR(LENGTH);
291 
292  uint8_t function = response[0];
293 
294  // Handle exception frames
295  if (function & 0x80 && responseLength == 2)
296  {
297  if (status->exceptionCallback)
298  status->exceptionCallback(
299  status,
300  address,
301  function & 0x7f,
302  (ModbusExceptionCode) response[1]);
303 
304  return MODBUS_NO_ERROR();
305  }
306 
307  // Check if function code matches the one in request frame
308  if (function != request[0])
309  return MODBUS_RESPONSE_ERROR(FUNCTION);
310 
311  // Find a parsing function
312  for (uint16_t i = 0; i < status->functionCount; i++)
313  if (function == status->functions[i].id)
314  {
315  ModbusErrorInfo err = status->functions[i].ptr(
316  status,
317  function,
318  request,
319  requestLength,
320  response,
321  responseLength);
322 
323  if (!modbusIsOk(err)) return err;
324  }
325 
326  // No matching function handler
327  return MODBUS_GENERAL_ERROR(FUNCTION);
328 }

344  ModbusMaster *status,
345  const uint8_t *request,
346  uint16_t requestLength,
347  const uint8_t *response,

349 {

394 LIGHTMODBUS_RET_ERROR modbusParseResponseTCP(
395  ModbusMaster *status,
396  const uint8_t *request,
397  uint16_t requestLength,
398  const uint8_t *response,
399  uint16_t responseLength)
400 {
401  // Check lengths
402  if (requestLength < 8) return MODBUS_REQUEST_ERROR(LENGTH);
403  if (responseLength < 8) return MODBUS_RESPONSE_ERROR(LENGTH);
404 
405  // Check if protocol IDs are correct
406  if (modbusRBE(&request[2]) != 0) return MODBUS_REQUEST_ERROR(BAD_PROTOCOL);
407  if (modbusRBE(&response[2]) != 0) return MODBUS_RESPONSE_ERROR(BAD_PROTOCOL);
408 
409  // Check if transaction IDs match
410  if (modbusRBE(&request[0]) != modbusRBE(&response[0]))
411  return MODBUS_RESPONSE_ERROR(BAD_TRANSACTION);
412 
413  // Check if lengths are ok
414  // The declared length includes the Unit ID byte (hence 6 and not 7 bytes are subtracted)
415  if (modbusRBE(&request[4]) != requestLength - 6) return MODBUS_REQUEST_ERROR(LENGTH);
416  if (modbusRBE(&response[4]) != responseLength - 6) return MODBUS_RESPONSE_ERROR(LENGTH);
417 
418  // Check if returned address is the same as in request
419  uint8_t address = response[6];
420  if (address != request[6])
421  return MODBUS_RESPONSE_ERROR(ADDRESS);
422 
423  return modbusParseResponsePDU(
424  status,
425  address,
426  request + 7,
427  requestLength - 7,
428  response + 7,
429  responseLength - 7);
